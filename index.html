function DNAStrand(dna){
  let dictionary = {
    'A': 'T',
    'T': 'A',
    'C': 'G',
    'G': 'C'
  }
  
  let result = '';
  
  for (let i = 0; i < dna.length; i++) {
    result += dictionary[dna[i]];
  }
  
  return result;
}
